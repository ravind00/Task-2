    <script>
        /* Form Validation & Alert [Source 84, 86] */
        const contactForm = document.getElementById('contactForm');

        contactForm.addEventListener('submit', (e) => {
            e.preventDefault();
            const email = document.getElementById('email').value;
            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

            if (!emailRegex.test(email)) {
                alert("❌ Error: Please enter a valid email format (e.g., name@domain.com).");
            } else {
                alert("✅ Success! Your message has been sent successfully.");
                contactForm.reset();
            }
        });

        /* DOM Manipulation - Add/Remove Task [Source 91, 92] */
        const taskInput = document.getElementById('taskInput');
        const addTaskBtn = document.getElementById('addTaskBtn');
        const todoList = document.getElementById('todo-list');

        addTaskBtn.addEventListener('click', () => {
            const taskValue = taskInput.value.trim();
            
            if (taskValue === "") {
                alert("Please type a task before adding.");
                return;
            }

            const li = document.createElement('li');
            li.className = 'todo-item';
            li.innerHTML = `
                <span>${taskValue}</span>
                <span class="delete-btn" title="Remove Task">Delete</span>
            `;

            li.querySelector('.delete-btn').addEventListener('click', () => {
                li.remove();
            });

            todoList.appendChild(li);
            taskInput.value = "";
        });
    </script>
